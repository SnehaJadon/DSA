
class Solution {
public:
    bool isPalindrome(ListNode* head) {
        vector<int> arr;

        ListNode* temp = head;
        while(temp){
            arr.push_back(temp->val);
            temp = temp-> next;
        }

        int start = 0, end = arr.size()-1;
        while (start <= end){
            if(arr[start] != arr[end]){
                return false;
            }
            start++;
            end--;
        }
        return true;
    }
};
